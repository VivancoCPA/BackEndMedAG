// --- Identidad y Estructura Familiar ---

Table users {
  id uuid [pk, default: `gen_random_uuid()`]
  full_name varchar
  email varchar [unique, not null]
  password_hash text [not null]
  photo_url text
  created_at timestamp [default: `now()`]
}

Table family_groups {
  id uuid [pk, default: `gen_random_uuid()`]
  name varchar [not null]
}

Table family_memberships {
  id serial [pk]
  user_id uuid [ref: > users.id]
  family_group_id uuid [ref: > family_groups.id]
  is_admin boolean [default: false]
  relationship varchar
}

// --- AnimalCare (Mascotas) ---

Table pets {
  id uuid [pk, default: `gen_random_uuid()`]
  family_group_id uuid [ref: > family_groups.id]
  name varchar [not null]
  species varchar
  breed varchar
  birth_date date
  photo_url text
}

// --- Entidades de Salud y Especialidades ---

Table specialties {
  id serial [pk]
  name varchar [not null, unique, note: 'Ej: Pediatría, Cardiología, Veterinaria']
  is_active boolean [default: true]
  created_at timestamp [default: `now()`]
}

Table doctors {
  id uuid [pk, default: `gen_random_uuid()`]
  name varchar [not null]
  last_name varchar [not null]
  specialty_id int [ref: > specialties.id] // FK a la nueva tabla
  register varchar [note: 'Número de colegiatura o registro médico']
  phone varchar
  email varchar
  photo_url text
  is_vet boolean [default: false]
  is_active boolean [default: true]
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
}

Table medical_centers {
  id uuid [pk, default: `gen_random_uuid()`]
  name varchar [not null]
  type varchar [note: 'Hospital, Clínica, Veterinaria']
  address text
  latitude float
  longitude float
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
}

Table doctor_affiliations {
  id serial [pk]
  doctor_id uuid [ref: > doctors.id]
  center_id uuid [ref: > medical_centers.id]
  office_number varchar
  work_schedule text
  
  Indexes {
    (doctor_id, center_id) [unique]
  }
}

// --- Historia Clínica Heredada ---

Table medical_incidents {
  id uuid [pk, default: `gen_random_uuid()`]
  owner_user_id uuid [ref: > users.id]
  pet_id uuid [ref: > pets.id]
  doctor_id uuid [ref: > doctors.id]
  center_id uuid [ref: > medical_centers.id]
  incident_date timestamp
  diagnosis text
  symptoms text
  emergency_level int
}

Table medical_exams {
  id uuid [pk]
  incident_id uuid [ref: > medical_incidents.id]
  exam_name varchar
  status varchar
  results_summary text
}

Table centers_type
{
  id int [pk]
  name varchar
  is_active boolean [default: true]
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
}

Table medications {
  id uuid [pk]
  incident_id uuid [ref: > medical_incidents.id]
  drug_name varchar
  posology text
  start_date date
  end_date date
}

// --- Evidencias y Otros ---

Table attachments {
  id uuid [pk]
  incident_id uuid [ref: > medical_incidents.id]
  exam_id uuid [ref: > medical_exams.id]
  file_url text
  file_type varchar
}

Table insurers {
  id uuid [pk]
  name varchar [not null]
  address varchar [not null]
  phone varchar [not null]
  email varchar [not null]
  person_in_charge varchar
  logo_url text
  is_active boolean [default: true]
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
}

Table user_pet_insurances {
  id uuid [pk]
  user_id uuid [ref: > users.id]
  pet_id uuid [ref: > pets.id]
  insurer_id uuid [ref: > insurers.id]
  policy_number varchar
  is_active boolean [default: true]
}
